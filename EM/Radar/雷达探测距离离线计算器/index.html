    雷达探测距离计算器¶

    在线计算器：¶

    https://claude.site/artifacts/62bcc1ce-bfa9-4db7-a6fc-af9c848432fa

    离线计算器¶

    要离线运行这个雷达探测距离计算器，您需要设置一个 React 项目并安装必要的依赖。我来详细说明步骤：

    1. 首先确保您的电脑已安装 Node.js，可以从官网下载：https://nodejs.org/

    2. 创建一个新的 Next.js 项目（这个项目使用了 Next.js 框架）： 

      npx create-next-app@latest radar-calculator
cd radar-calculator
      在提示时选择：

    3. TypeScript: No
    4. ESLint: Yes
    5. Tailwind CSS: Yes

    6. 其他选项可以根据需要选择

    7. 安装必要的依赖： 

      npm install @radix-ui/react-icons @radix-ui/react-label @radix-ui/react-separator @radix-ui/react-slot class-variance-authority clsx tailwind-merge

    8. 安装 shadcn/ui 组件： 

      npx shadcn-ui@latest init

    然后安装用到的组件： 

    npx shadcn-ui@latest add card input label separator

    1. 在 src/app/page.js 中替换内容为： 

      import RangeCalculator from '../components/RangeCalculator';

export default function Home() {
  return (
    <main className="container mx-auto p-4">
      <RangeCalculator />
    </main>
  );
}

    2. 创建新文件 src/components/RangeCalculator.jsx，将我之前提供的计算器代码复制到这个文件中。

    3. 运行开发服务器： 

      npm run dev

    访问 http://localhost:3000 就能看到计算器了。

    如果您想部署到生产环境： 

    npm run build
npm run start

    项目文件结构应该如下： 

    radar-calculator/
├── node_modules/
├── public/
├── src/
│   ├── app/
│   │   ├── page.js
│   │   └── layout.js
│   └── components/
│       └── RangeCalculator.jsx
├── package.json
└── tailwind.config.js

    一些注意事项： 1. 确保您的 tailwind.config.js 包含了所有必要的配置 2. 如果遇到样式问题，可能需要在 globals.css 中添加相应的 Tailwind 导入 3. 如果需要更好的性能，可以考虑添加状态管理（如 Redux）和缓存机制

    计算器代码¶

    干扰机类型相同¶

    import React, { useState } from 'react';
import { Card, CardHeader, CardTitle, CardContent } from '@/components/ui/card';
import { Input } from '@/components/ui/input';
import { Label } from '@/components/ui/label';
import { Separator } from '@/components/ui/separator';

const RangeCalculator = () => {
  // dB to linear conversion
  const dBToLinear = (db) => Math.pow(10, db/10);
  // Linear to dB conversion
  const linearTodB = (linear) => 10 * Math.log10(linear);

  const [params, setParams] = useState({
    // 雷达参数
    Pt: 1000,    // 雷达发射机峰值功率 (W)
    Gt_dB: 30,   // 雷达天线主瓣增益 (dB)
    Gs_dB: -20,  // 雷达天线副瓣增益 (dB)

    // 目标参数
    sigma: 1,    // 目标雷达截面积 (m²)

    // 信号处理参数
    Kj: 0.5,     // 干扰抑制系数
    D1: 100,     // 脉冲压缩处理增益
    D2: 10,      // MTD处理增益
    Dav: 0.8,    // 平均跨接和遮挡损耗
    Lt_dB: -1.5, // 雷达系统损耗 (dB)
    Kf: 1,       // 带宽比
    rj: 1        // 极化损耗
  });

  // 干扰机参数数组
  const [jammers, setJammers] = useState([
    {
      Rj: 10000,    // 距离 (m)
      Gs_dB: -20,   // 雷达天线副瓣增益 (dB)
      Pj: 100,      // 发射功率 (W)
      Gj_dB: 10     // 天线增益 (dB)
    }
  ]);

  // 计算综合增益D
  const D = params.D1 * params.D2 * params.Dav;

  // 添加新的干扰机
  const addJammer = () => {
    setJammers([...jammers, {
      Rj: 10000,
      Gs_dB: -20,
      Pj: 100,
      Gj_dB: 10
    }]);
  };

  // 删除干扰机
  const removeJammer = (index) => {
    if (jammers.length > 1) {
      const newJammers = jammers.filter((_, i) => i !== index);
      setJammers(newJammers);
    }
  };

  // 计算K值
  const calculateK = (jammer) => {
    const Gt = dBToLinear(params.Gt_dB);
    const Lt = dBToLinear(params.Lt_dB);
    const Gj = dBToLinear(jammer.Gj_dB);

    return (params.Pt * Gt * params.sigma * params.Kj * D * Lt) / 
           (4 * Math.PI * jammer.Pj * Gj * params.Kf * params.rj);
  };

  // 计算探测距离
  const calculateRange = () => {
    const denominatorSum = jammers.reduce((sum, jammer) => {
      const Gs = dBToLinear(jammer.Gs_dB);
      return sum + Gs / Math.pow(jammer.Rj, 2);
    }, 0);

    const K = calculateK(jammers[0]); // 使用第一个干扰机的参数计算K
    return Math.pow(K / denominatorSum, 0.25);
  };

  const InputGroup = ({ label, value, onChange, unit = '' }) => (
    <div className="space-y-1">
      <Label className="text-sm">{label}</Label>
      <Input
        type="number"
        value={value}
        onChange={e => onChange(parseFloat(e.target.value))}
        className="w-full"
      />
      {unit && <span className="text-xs text-gray-500 ml-1">{unit}</span>}
    </div>
  );

  return (
    <Card className="w-full max-w-6xl">
      <CardHeader>
        <CardTitle>雷达探测距离计算器</CardTitle>
      </CardHeader>
      <CardContent>
        <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
          <div className="space-y-6">
            <div>
              <h3 className="text-lg font-semibold mb-4">雷达参数</h3>
              <div className="grid grid-cols-2 gap-4">
                <InputGroup 
                  label="发射机峰值功率 (Pt)"
                  value={params.Pt}
                  onChange={val => setParams({...params, Pt: val})}
                  unit="W"
                />
                <InputGroup 
                  label="天线主瓣增益 (Gt)"
                  value={params.Gt_dB}
                  onChange={val => setParams({...params, Gt_dB: val})}
                  unit="dB"
                />
              </div>
            </div>

            <div>
              <h3 className="text-lg font-semibold mb-4">目标参数</h3>
              <div className="grid grid-cols-2 gap-4">
                <InputGroup 
                  label="雷达截面积 (σ)"
                  value={params.sigma}
                  onChange={val => setParams({...params, sigma: val})}
                  unit="m²"
                />
              </div>
            </div>

            <div>
              <h3 className="text-lg font-semibold mb-4">信号处理参数</h3>
              <div className="grid grid-cols-2 gap-4">
                <InputGroup 
                  label="脉冲压缩增益 (D1)"
                  value={params.D1}
                  onChange={val => setParams({...params, D1: val})}
                />
                <InputGroup 
                  label="MTD增益 (D2)"
                  value={params.D2}
                  onChange={val => setParams({...params, D2: val})}
                />
                <InputGroup 
                  label="平均损耗 (Dav)"
                  value={params.Dav}
                  onChange={val => setParams({...params, Dav: val})}
                />
                <InputGroup 
                  label="系统损耗 (Lt)"
                  value={params.Lt_dB}
                  onChange={val => setParams({...params, Lt_dB: val})}
                  unit="dB"
                />
              </div>
            </div>
          </div>

          <div className="space-y-6">
            <div>
              <div className="flex justify-between items-center mb-4">
                <h3 className="text-lg font-semibold">干扰机参数</h3>
                <button 
                  onClick={addJammer}
                  className="px-3 py-1 bg-blue-500 text-white rounded hover:bg-blue-600"
                >
                  添加干扰机
                </button>
              </div>

              {jammers.map((jammer, index) => (
                <div key={index} className="mb-6 p-4 border rounded">
                  <div className="flex justify-between items-center mb-4">
                    <h4 className="font-semibold">干扰机 {index + 1}</h4>
                    {jammers.length > 1 && (
                      <button 
                        onClick={() => removeJammer(index)}
                        className="px-2 py-1 bg-red-500 text-white rounded hover:bg-red-600"
                      >
                        删除
                      </button>
                    )}
                  </div>
                  <div className="grid grid-cols-2 gap-4">
                    <InputGroup 
                      label="距离 (Rj)"
                      value={jammer.Rj}
                      onChange={val => {
                        const newJammers = [...jammers];
                        newJammers[index].Rj = val;
                        setJammers(newJammers);
                      }}
                      unit="m"
                    />
                    <InputGroup 
                      label="副瓣增益 (Gs)"
                      value={jammer.Gs_dB}
                      onChange={val => {
                        const newJammers = [...jammers];
                        newJammers[index].Gs_dB = val;
                        setJammers(newJammers);
                      }}
                      unit="dB"
                    />
                    <InputGroup 
                      label="发射功率 (Pj)"
                      value={jammer.Pj}
                      onChange={val => {
                        const newJammers = [...jammers];
                        newJammers[index].Pj = val;
                        setJammers(newJammers);
                      }}
                      unit="W"
                    />
                    <InputGroup 
                      label="天线增益 (Gj)"
                      value={jammer.Gj_dB}
                      onChange={val => {
                        const newJammers = [...jammers];
                        newJammers[index].Gj_dB = val;
                        setJammers(newJammers);
                      }}
                      unit="dB"
                    />
                  </div>
                </div>
              ))}
            </div>
          </div>
        </div>

        <Separator className="my-6" />

        <div className="text-center">
          <h3 className="text-xl font-bold mb-2">计算结果</h3>
          <p className="text-3xl font-semibold text-blue-600">
            最大探测距离: {calculateRange().toFixed(2)} m
          </p>
        </div>
      </CardContent>
    </Card>
  );
};

export default RangeCalculator;

    主要更改包括：

    1. 增益单位转换：
    2. 添加了 dBToLinear() 和 linearTodB() 函数进行 dB 和线性值的转换

    3. 在计算中使用线性值，在界面显示 dB 值

    4. 参数更新：

    5. 雷达天线主瓣增益 (Gt): 改为 dB
    6. 雷达天线副瓣增益 (Gs): 改为 dB
    7. 干扰机天线增益 (Gj): 改为 dB

    8. 系统损耗 (Lt): 改为 dB

    9. 默认值调整：

    10. 主瓣增益默认值设为 30 dB
    11. 副瓣增益默认值设为 -20 dB
    12. 系统损耗默认值设为 -1.5 dB
    13. 干扰机天线增益默认值设为 10 dB

    计算过程中，所有 dB 值都会先转换为线性值进行计算，最终显示的探测距离不变。这样的修改更符合实际工程应用中的使用习惯，因为雷达系统中的增益通常都是以 dB 为单位给出的。

  const dBToLinear = (db) => Math.pow(10, db/10);

  const linearTodB = (linear) => 10 * Math.log10(linear);

    // 信号处理参数
    Kj: 0.5,      // 干扰抑制系数
    D1_dB: 20,    // 脉冲压缩处理增益 (dB)
    D2_dB: 10,    // MTD处理增益 (dB)
    Dav_dB: -1,   // 平均跨接和遮挡损耗 (dB)
    Lt_dB: -1.5,  // 雷达系统损耗 (dB)
    Kf: 1,        // 带宽比
    rj: 1         // 极化损耗
  });

  const [jammers, setJammers] = useState([

      Gj_dB: 10     // 天线增益 (dB)

  ]);

  const calculateD = () => {
    const D1 = dBToLinear(params.D1_dB);
    const D2 = dBToLinear(params.D2_dB);
    const Dav = dBToLinear(params.Dav_dB);
    return D1 * D2 * Dav;
  };

  // 添加新的干扰机
  const addJammer = () => {

    const denominatorSum = jammers.reduce((sum, jammer) => {
      const Gs = dBToLinear(jammer.Gs_dB);
      return sum + Gs / Math.pow(jammer.Rj, 2);
    }, 0);

    const K = calculateK(jammers[0]);
    return Math.pow(K / denominatorSum, 0.25);
  };

  const InputGroup = ({ label, value, onChange, unit = '' }) => (
    <div className="space-y-1">
      <Label className="text-sm">{label}</Label>
      <Input
        type="number"
        value={value}
        onChange={e => onChange(parseFloat(e.target.value))}
        className="w-full"
      />
      {unit && <span className="text-xs text-gray-500 ml-1">{unit}</span>}

      <CardHeader>
        <CardTitle>雷达探测距离计算器</CardTitle>
      </CardHeader>
      <CardContent>

              <h3 className="text-lg font-semibold mb-4">雷达参数</h3>

                  label="发射机峰值功率 (Pt)"
                  value={params.Pt}
                  onChange={val => setParams({...params, Pt: val})}
                  unit="W"

                  label="天线主瓣增益 (Gt)"
                  value={params.Gt_dB}
                  onChange={val => setParams({...params, Gt_dB: val})}
                  unit="dB"
                />
              </div>
            </div>

            <div>
              <h3 className="text-lg font-semibold mb-4">目标参数</h3>
              <div className="grid grid-cols-2 gap-4">
                <InputGroup 
                  label="雷达截面积 (σ)"
                  value={params.sigma}
                  onChange={val => setParams({...params, sigma: val})}
                  unit="m²"
                />
              </div>
            </div>

            <div>
              <h3 className="text-lg font-semibold mb-4">信号处理参数</h3>
              <div className="grid grid-cols-2 gap-4">
                <InputGroup 
                  label="脉冲压缩增益 (D1)"
                  value={params.D1_dB}
                  onChange={val => setParams({...params, D1_dB: val})}
                  unit="dB"
                />
                <InputGroup 
                  label="MTD增益 (D2)"
                  value={params.D2_dB}

                  unit="dB"
                />
                <InputGroup 
                  label="系统损耗 (Lt)"
                  value={params.Lt_dB}
                  onChange={val => setParams({...params, Lt_dB: val})}
                  unit="dB"
                />
              </div>
            </div>

            <div>
              <h3 className="text-lg font-semibold mb-4">综合增益</h3>
              <p className="text-lg">
                总增益: {linearTodB(calculateD()).toFixed(2)} dB

    import React, { useState } from 'react';
import { Card, CardHeader, CardTitle, CardContent } from '@/components/ui/card';
import { Input } from '@/components/ui/input';
import { Label } from '@/components/ui/label';
import { Separator } from '@/components/ui/separator';

const RangeCalculator = () => {
  // dB to linear conversion
  const dBToLinear = (db) => Math.pow(10, db/10);
  // Linear to dB conversion
  const linearTodB = (linear) => 10 * Math.log10(linear);

  const [params, setParams] = useState({
    // 雷达参数
    Pt: 1000,     // 雷达发射机峰值功率 (W)
    Gt_dB: 30,    // 雷达天线主瓣增益 (dB)

    // 目标参数
    sigma: 1,     // 目标雷达截面积 (m²)

    // 信号处理参数
    Kj: 0.5,      // 干扰抑制系数
    D1_dB: 20,    // 脉冲压缩处理增益 (dB)
    D2_dB: 10,    // MTD处理增益 (dB)
    Dav_dB: -1,   // 平均跨接和遮挡损耗 (dB)
    Lt_dB: -1.5,  // 雷达系统损耗 (dB)
  });

  // 干扰机参数数组，每个干扰机都有独立的参数
  const [jammers, setJammers] = useState([
    {
      Rj: 10000,    // 距离 (m)
      Gs_dB: -20,   // 雷达天线副瓣增益 (dB)
      Pj: 100,      // 发射功率 (W)
      Gj_dB: 10,    // 天线增益 (dB)
      Kf: 1.0,      // 带宽比
      rj: 1.0       // 极化损耗
    }
  ]);

  // 计算综合增益D
  const calculateD = () => {
    const D1 = dBToLinear(params.D1_dB);
    const D2 = dBToLinear(params.D2_dB);
    const Dav = dBToLinear(params.Dav_dB);
    return D1 * D2 * Dav;
  };

  // 添加新的干扰机
  const addJammer = () => {
    setJammers([...jammers, {
      Rj: 10000,
      Gs_dB: -20,
      Pj: 100,
      Gj_dB: 10,
      Kf: 1.0,
      rj: 1.0
    }]);
  };

  // 删除干扰机
  const removeJammer = (index) => {
    if (jammers.length > 1) {
      const newJammers = jammers.filter((_, i) => i !== index);
      setJammers(newJammers);
    }
  };

  // 计算探测距离
  const calculateRange = () => {
    const Gt = dBToLinear(params.Gt_dB);
    const Lt = dBToLinear(params.Lt_dB);
    const D = calculateD();

    // 计算分子
    const numerator = params.Pt * Gt * params.sigma * params.Kj * D * Lt;

    // 计算分母中的求和项
    const denominatorSum = jammers.reduce((sum, jammer) => {
      const Gs = dBToLinear(jammer.Gs_dB);
      const Gj = dBToLinear(jammer.Gj_dB);
      return sum + (jammer.Pj * Gj * Gs) / (Math.pow(jammer.Rj, 2) * jammer.Kf * jammer.rj);
    }, 0);

    // 计算最终结果
    return Math.pow(numerator / (4 * Math.PI * denominatorSum), 0.25);
  };

  const InputGroup = ({ label, value, onChange, unit = '' }) => (
    <div className="space-y-1">
      <Label className="text-sm">{label}</Label>
      <Input
        type="number"
        value={value}
        onChange={e => onChange(parseFloat(e.target.value))}
        className="w-full"
      />
      {unit && <span className="text-xs text-gray-500 ml-1">{unit}</span>}
    </div>
  );

  return (
    <Card className="w-full max-w-6xl">
      <CardHeader>
        <CardTitle>多类型干扰机雷达探测距离计算器</CardTitle>
      </CardHeader>
      <CardContent>
        <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
          <div className="space-y-6">
            <div>
              <h3 className="text-lg font-semibold mb-4">雷达参数</h3>
              <div className="grid grid-cols-2 gap-4">
                <InputGroup 
                  label="发射机峰值功率 (Pt)"
                  value={params.Pt}
                  onChange={val => setParams({...params, Pt: val})}
                  unit="W"
                />
                <InputGroup 
                  label="天线主瓣增益 (Gt)"
                  value={params.Gt_dB}
                  onChange={val => setParams({...params, Gt_dB: val})}
                  unit="dB"
                />
              </div>
            </div>

            <div>
              <h3 className="text-lg font-semibold mb-4">目标参数</h3>
              <div className="grid grid-cols-2 gap-4">
                <InputGroup 
                  label="雷达截面积 (σ)"
                  value={params.sigma}
                  onChange={val => setParams({...params, sigma: val})}
                  unit="m²"
                />
              </div>
            </div>

            <div>
              <h3 className="text-lg font-semibold mb-4">信号处理参数</h3>
              <div className="grid grid-cols-2 gap-4">
                <InputGroup 
                  label="脉冲压缩增益 (D1)"
                  value={params.D1_dB}
                  onChange={val => setParams({...params, D1_dB: val})}
                  unit="dB"
                />
                <InputGroup 
                  label="MTD增益 (D2)"
                  value={params.D2_dB}
                  onChange={val => setParams({...params, D2_dB: val})}
                  unit="dB"
                />
                <InputGroup 
                  label="平均损耗 (Dav)"
                  value={params.Dav_dB}
                  onChange={val => setParams({...params, Dav_dB: val})}
                  unit="dB"
                />
                <InputGroup 
                  label="系统损耗 (Lt)"
                  value={params.Lt_dB}
                  onChange={val => setParams({...params, Lt_dB: val})}
                  unit="dB"
                />
                <InputGroup 
                  label="干扰抑制系数 (Kj)"
                  value={params.Kj}
                  onChange={val => setParams({...params, Kj: val})}
                />
              </div>
            </div>

            <div>
              <h3 className="text-lg font-semibold mb-4">综合增益</h3>
              <p className="text-lg">
                总增益: {linearTodB(calculateD()).toFixed(2)} dB
              </p>
            </div>
          </div>

          <div className="space-y-6">
            <div>
              <div className="flex justify-between items-center mb-4">
                <h3 className="text-lg font-semibold">干扰机参数</h3>
                <button 
                  onClick={addJammer}
                  className="px-3 py-1 bg-blue-500 text-white rounded hover:bg-blue-600"
                >
                  添加干扰机
                </button>
              </div>

              {jammers.map((jammer, index) => (
                <div key={index} className="mb-6 p-4 border rounded">
                  <div className="flex justify-between items-center mb-4">
                    <h4 className="font-semibold">干扰机 {index + 1}</h4>
                    {jammers.length > 1 && (

                        className="px-2 py-1 bg-red-500 text-white rounded hover:bg-red-600"
                      >
                        删除
                      </button>
                    )}
                  </div>
                  <div className="grid grid-cols-2 gap-4">
                    <InputGroup 
                      label="距离 (Rj)"
                      value={jammer.Rj}
                      onChange={val => {
                        const newJammers = [...jammers];
                        newJammers[index].Rj = val;
                        setJammers(newJammers);
                      }}
                      unit="m"
                    />
                    <InputGroup 
                      label="副瓣增益 (Gs)"
                      value={jammer.Gs_dB}

                        const newJammers = [...jammers];
                        newJammers[index].Gs_dB = val;
                        setJammers(newJammers);
                      }}
                      unit="dB"
                    />
                    <InputGroup

                    <InputGroup

        <Separator className="my-6" />

        <div className="text-center">
          <h3 className="text-xl font-bold mb-2">计算结果</h3>
